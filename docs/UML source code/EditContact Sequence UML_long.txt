title EditContact

actor User
participant ":LogicManager" as LogicManager #pink
participant ":CommandHandler" as CommandHandler #lightgreen
participant "Contact" as Contact #lightyellow

User->LogicManager:edit contact
activate LogicManager
LogicManager->CommandHandler:editContact(time slot)
activate CommandHandler

opt mainUser
alt valid edit
CommandHandler->Contact:isValidEdit(time slot)
activate Contact
CommandHandler<--Contact:valid edit
deactivate Contact


else invalid edit
CommandHandler->Contact:isValidEdit(time slot)
activate Contact
CommandHandler<--Contact:invalid edit exception
deactivate Contact


LogicManager<--CommandHandler:invalid edit exception
deactivate CommandHandler
User<--LogicManager:inform user of invalid edit
end
activate CommandHandler
end

CommandHandler->Contact:editSchedule(time slot)
activate Contact
CommandHandler<--Contact:
deactivate Contact
LogicManager<--CommandHandler:
deactivate CommandHandler
User<--LogicManager:
deactivate LogicManager
